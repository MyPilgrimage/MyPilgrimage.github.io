<script>
function progress-move() {
  var progressBar = document.getElementById("progress-route");
  console.log("width: "+ progressBar.width);
  var width = progressBar.width;
  var interval = setInterval(frame, 10);
  function frame() {
    if (width >= 100) {
      clearInterval(interval);
    } else {
      width++;
      progressBar.style.width = width + '%';
	  progressBar.innerHTML = width * 1  + '%';
    }
  }
}
</script>
